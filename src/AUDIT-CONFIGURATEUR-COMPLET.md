```typescript
interface ScoringWeights {
  painPoint: number;
  futureVision: number;
  decisionStyle: number;
  household: number;

interface ConfigScore {
  model: string;
  finish: string;
  capacity: string;
  waterModule: string;
  subscription: string;
  score: number; // 0-100
  reasoning: string[];
}

const scoreConfiguration = (
  profile: UserPsychoProfile,
  config: Config
): number => {
  let score = 0;
  
  // 1. Score Pain Point (30%)
  if (profile.painPoint === 'time' && config.waterModule !== 'none') score += 30;
  if (profile.painPoint === 'money' && config.subscription === 'none') score += 30;
  if (profile.painPoint === 'ecology' && config.waterModule === 'cube') score += 30;
  if (profile.painPoint === 'health' && config.waterModule !== 'none') score += 30;
  if (profile.painPoint === 'aesthetics' && ['champagne', 'black-matt'].includes(config.finish)) score += 30;
  
  // 2. Score Future Vision (25%)
  if (profile.futureVision === 'time-optimizer' && config.capacity === 'combi') score += 25;
  if (profile.futureVision === 'money-saver' && config.subscription === 'none') score += 25;
  if (profile.futureVision === 'responsible-parent' && config.waterModule === 'filter') score += 25;
  if (profile.futureVision === 'eco-warrior' && config.waterModule === 'cube') score += 25;
  if (profile.futureVision === 'social-showcase' && ['champagne', 'black-matt'].includes(config.finish)) score += 25;
  
  // 3. Score Decision Style (20%)
  if (profile.decisionStyle === 'analytical') {
    // Analytique : favoriser configs avec meilleur ROI
    const roi = calculateROI(config);
    score += (roi / 12) * 20; // ROI 6 mois = 20 pts
  }
  if (profile.decisionStyle === 'emotional') {
    // Émotionnel : favoriser finitions premium + témoignages
    if (['champagne', 'black-matt', 'gold'].includes(config.finish)) score += 20;
  }
  if (profile.decisionStyle === 'value-seeker') {
    // Value : favoriser configs économiques
    if (config.subscription === 'none' || config.subscription === 'filter-only') score += 20;
  }
  
  // 4. Score Household (15%)
  if (profile.household === 'famille-plus' && config.capacity === 'combi') score += 15;
  if (profile.household === 'solo' && config.capacity === 'pro3') score += 15;
  
  // 5. Bonus cohérence (10%)
  if (isCoherent(config)) score += 10;
  
  return Math.min(score, 100);
};

// Générer TOP 3 configs
const generateTopConfigs = (profile: UserPsychoProfile): ConfigScore[] => {
  const allCombinations = generateAllConfigs();
  const scored = allCombinations.map(config => ({
    ...config,
    score: scoreConfiguration(profile, config),
    reasoning: generateReasoning(profile, config)
  }));
  
  return scored
    .sort((a, b) => b.score - a.score)
    .slice(0, 3);
};
```

```typescript
  const [variant, setVariant] = useState<string>('A');

    // Récupérer variant depuis localStorage ou assigner aléatoirement
    const stored = localStorage.getItem(`ab_${testId}`);

    } else {
      const test = Object.values(AB_TESTS).find(t => t.id === testId);

  }, [testId]);
```
